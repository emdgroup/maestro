3. After the worktree remove call, if `delete_branch` is true, delete the branch (best-effort, non-fatal — follow the pattern from review_handlers.rs line 301-320):
   ```rust
   if delete_branch {
       // Best-effort branch deletion — don't fail the whole operation
       match &git_conn {
           crate::models::GitConnection::Local { path } => {
               let output = tokio::process::Command::new("git")
                   .args(["branch", "-d", &wt_branch_name])
                   .current_dir(path)
                   .output()
                   .await;
               match output {
                   Ok(o) if !o.status.success() => {
                       let stderr = String::from_utf8_lossy(&o.stderr);
                       eprintln!("delete_worktree: branch delete failed (non-fatal): {}", stderr);
                   }
                   Err(e) => {
                       eprintln!("delete_worktree: branch delete error (non-fatal): {}", e);
                   }
                   _ => {
                       eprintln!("delete_worktree: branch {} deleted", wt_branch_name);
                   }
               }
           }
           crate::models::GitConnection::Remote { ssh, remote_path } => {
               let cmd = format!("git -C {} branch -d {}", crate::git::remote::shell_quote(remote_path), crate::git::remote::shell_quote(&wt_branch_name));
               let _ = ssh.execute_command(&cmd).await;
               eprintln!("delete_worktree: remote branch delete attempted for {}", wt_branch_name);
           }
       }
   }
   ```

   Use `-d` (lowercase) not `-D` — this is a safe delete that refuses to delete unmerged branches.
