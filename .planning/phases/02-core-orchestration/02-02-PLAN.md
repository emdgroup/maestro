  <action>
    1. Open src-tauri/src/ipc/handlers.rs (or create if missing from Phase 1)
    2. Add create_task handler function:
       ```rust
       #[tauri::command]
       pub fn create_task(
           state: State<Arc<AppState>>,
           project_id: i32,
           name: String,
           description: String,
           acceptance_criteria: String,
           skills: Vec<String>,
       ) -> Result<Task, String> {
           // Validate inputs
           if name.trim().is_empty() || name.len() < 3 || name.len() > 255 {
               return Err("Name must be 3-255 characters".to_string());
           }
           if description.trim().is_empty() || description.len() < 10 {
               return Err("Description must be at least 10 characters".to_string());
           }
           if acceptance_criteria.trim().is_empty() || acceptance_criteria.len() < 10 {
               return Err("Acceptance criteria must be at least 10 characters".to_string());
           }

           let conn = state.db.lock()
               .map_err(|e| format!("Lock failed: {}", e))?;

           let created_at = chrono::Local::now().to_rfc3339();
           let updated_at = created_at.clone();
           let skills_json = serde_json::to_string(&skills)
               .map_err(|e| format!("JSON serialization failed: {}", e))?;

           conn.execute(
               "INSERT INTO tasks (project_id, name, description, acceptance_criteria, skills, status, created_at, updated_at)
                VALUES (?1, ?2, ?3, ?4, ?5, 'backlog', ?6, ?7)",
               rusqlite::params![project_id, name, description, acceptance_criteria, skills_json, created_at, updated_at],
           )
           .map_err(|e| format!("Insert failed: {}", e))?;

           let task_id = conn.last_insert_rowid() as i32;

           // Fetch and return created task
           let mut stmt = conn.prepare(
               "SELECT id, project_id, name, description, acceptance_criteria, skills, status, external_id, is_imported, import_source, created_at, updated_at
                FROM tasks WHERE id = ?1"
           )
           .map_err(|e| e.to_string())?;

           let task = stmt.query_row([task_id], |row| {
               Ok(Task {
                   id: row.get(0)?,
                   project_id: row.get(1)?,
                   name: row.get(2)?,
                   description: row.get(3)?,
                   acceptance_criteria: row.get(4)?,
                   skills: serde_json::from_str(&row.get::<_, String>(5)?).unwrap_or_default(),
                   status: row.get(6)?,
                   external_id: row.get(7)?,
                   is_imported: row.get(8)?,
                   import_source: row.get(9)?,
                   created_at: row.get(10)?,
                   updated_at: row.get(11)?,
               })
           })
           .map_err(|e| e.to_string())?;

           Ok(task)
       }
       ```
    3. Verify database schema has skills column (text, storing JSON array):

    4. Ensure handler is exported in main.rs tauri::Builder::default().invoke_handler()

    Note: Validation mirrors frontend requirements (min/max length). Skills stored as JSON array in TEXT column. Handler uses AppState pattern from Phase 01-04.
