#[tauri::command]
pub async fn pause_agent_execution(
  state: State<Arc<AppState>>,
  task_id: i32,
) -> Result<(), String> {
  let conn = state.db.lock().map_err(|e| format!("Failed to lock DB: {}", e))?;

  // Get current execution log for this task
  let exec_log = get_current_execution_log(&conn, task_id)
    .map_err(|e| format!("Failed to get execution log: {}", e))?;

  // Send SIGSTOP to running process (Phase 8-02 pattern)
  // get_process_handle(exec_log.id) returns Option<ProcessHandle>
  if let Some(handle) = get_process_handle(exec_log.id) {
    handle.pause().map_err(|e| format!("Failed to pause process: {}", e))?;
  }

  // Update execution log status to Paused in database
  pause_execution_log(&conn, exec_log.id)
    .map_err(|e| format!("Failed to pause execution: {}", e))?;

  Ok(())
}
```

2. Add resume_agent_execution handler:

#[tauri::command]
pub async fn resume_agent_execution(
  state: State<Arc<AppState>>,
  task_id: i32,
  project_id: i32,
  repo_path: String,
) -> Result<i32, String> {
  // Step 1: Get current paused execution log
  let conn = state.db.lock().map_err(|e| format!("Failed to lock DB: {}", e))?;
  let prev_exec_log = get_current_execution_log(&conn, task_id)
    .map_err(|e| format!("Failed to get execution log: {}", e))?;

  // Step 2: Create new execution log (resume starts fresh log, not continuing old one)
  let new_exec_log = create_execution_log(&conn, task_id)
    .map_err(|e| format!("Failed to create new execution log: {}", e))?;
  let exec_id = new_exec_log.id;
  drop(conn);

  // Step 3: Reacquire lock before lease_worktree (CRITICAL FIX: lease_worktree expects &Connection, not &Mutex)
  let conn = state.db.lock().map_err(|e| format!("Failed to lock DB: {}", e))?;
  let worktree = lease_worktree(&conn, project_id, task_id, &repo_path).await?;
  let worktree_id = worktree.id;
  let worktree_path = worktree.path.clone();
  drop(conn);

  tokio::spawn(async move {
    let result = run_agent_process(&worktree_path, task_id).await;
    let conn = state.db.lock().unwrap();
    update_execution_log(&conn, exec_id, result).ok();
    return_worktree(&conn, worktree_id).ok();
  });

  Ok(new_exec_log.id)
}
```

pub fn pause_execution_log(conn: &Connection, exec_log_id: i32) -> Result<(), rusqlite::Error> {

pub fn get_current_execution_log(conn: &Connection, task_id: i32) -> Result<ExecutionLog, rusqlite::Error> {

}
