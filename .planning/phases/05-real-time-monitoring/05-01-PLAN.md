2. Implement `send_terminal_input` handler in handlers.rs:
   ```rust
   #[tauri::command]
   async fn send_terminal_input(
       state: tauri::State<'_, Arc<AppState>>,
       task_id: i32,
       input: String,
   ) -> Result<(), String> {
       let sessions = state.pty_sessions.lock().await;
       let session = sessions.get(&task_id)
           .ok_or_else(|| format!("No PTY session for task {}", task_id))?
           .clone();
       drop(sessions);

       let mut session = session.lock().await;
       session.write_input(input.as_bytes()).await
   }
   ```

3. Implement `resize_terminal` handler in handlers.rs:

   ```rust
   async fn resize_terminal(
       state: tauri::State<'_, Arc<AppState>>,
       task_id: i32,
       cols: u16,
       rows: u16,
   ) -> Result<(), String> {
       let sessions = state.pty_sessions.lock().await;
       let session = sessions.get(&task_id)
           .ok_or_else(|| format!("No PTY session for task {}", task_id))?
           .clone();
       drop(sessions);

       let mut session = session.lock().await;
       session.resize_pty(cols, rows).await
   }
   ```

4. Add write_input and resize_pty methods to PtySession:
   ```rust
   impl PtySession {
       pub async fn write_input(&mut self, data: &[u8]) -> Result<(), String> {
           use std::io::Write;
           self.master.write_all(data)
               .map_err(|e| e.to_string())
       }

       pub async fn resize_pty(&mut self, cols: u16, rows: u16) -> Result<(), String> {
           use portable_pty::PtySize;
           self.master.resize(PtySize {
               rows,
               cols,
               pixel_width: 0,
               pixel_height: 0,
           }).map_err(|e| e.to_string())
       }
   }
   ```
