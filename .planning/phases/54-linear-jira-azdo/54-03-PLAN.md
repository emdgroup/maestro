FUNCTION validate_and_store(project_id: i32, base_url: &str, project_key: &str, token: &str, project_path: &str, app_state: &crate::db::AppState) -> Result<String, String>:
- let base = normalize_instance_url(base_url);
- Build reqwest::Client with 15s timeout
- GET format!("{}/rest/api/2/myself", base) with header: Authorization: format!("Bearer {}", token), Accept: application/json
- HTTP error handling: 401 → "Jira Server: bad credentials"; other → "Jira Server API error {status}: {reason}"
- Deserialize as JiraServerMyselfResponse
- display_name = user.display_name.or(user.name).unwrap_or_else(|| "unknown".to_string())
- Save TicketingConfig { provider: Some(ProviderConfig::Jiraserver(JiraServerConfig { base_url: base.clone(), project_key: project_key.to_string() })), updated_at: now_rfc3339() }
- Store StoredToken { access_token: token.to_string(), refresh_token: None, expires_at: None, provider: "jira_server".to_string() }
- Return Ok(display_name)

FUNCTION fetch_issues(base_url: &str, project_key: &str, token: &str) -> Result<Vec<RemoteIssue>, String>:
- let base = normalize_instance_url(base_url);
- Build reqwest::Client with 15s timeout
- GET "{base}/rest/api/2/search" with .query(&[("jql", format!("project = {} AND statusCategory != Done ORDER BY updated DESC", project_key)), ("maxResults", "100".to_string()), ("fields", "summary,description,labels,updated".to_string())])

- HTTP error handling same pattern

  - title: issue.fields.summary
  - body: issue.fields.description  (plain string, no conversion needed)
  - url: format!("{}/browse/{}", base, issue.key)
  - labels: issue.fields.labels
  - updated_at: issue.fields.updated

FUNCTION fetch_issues(org_url: &str, project: &str, token: &str) -> Result<Vec<RemoteIssue>, String>:
- let base = normalize_instance_url(org_url);
- let auth = make_azdo_auth(token);
- Build reqwest::Client with 15s timeout
- Step 1: POST format!("{}/{}/_apis/wit/wiql?api-version=7.1", base, project) with body WiqlRequest { query: &wiql_query } where wiql_query = format!("SELECT [System.Id] FROM WorkItems WHERE [System.TeamProject] = '{}' AND [System.State] <> 'Closed'", project) and header Authorization: auth
- HTTP error handling same pattern
- Deserialize WIQL response as WiqlResponse; extract Vec<i32> of IDs via .map(|r| r.id).collect()
- If IDs is empty → return Ok(vec![])
- Step 2: Chunk IDs into slices of 200 with .chunks(200); for each chunk POST format!("{}/{}/_apis/wit/workitemsbatch?api-version=7.1", base, project) with body BatchRequest { ids: chunk, fields: WIQL_FIELDS } and header Authorization: auth
- HTTP error handling for each batch call
- Deserialize each batch as BatchResponse; extend a results Vec<RemoteIssue>
- For each WorkItemDetail map to RemoteIssue:
  - external_id: format!("azuredevops:{}", item.id)
  - title: item.fields.title
  - body: item.fields.description  (HTML as-is for v1.6 — no HTML stripping per RESEARCH.md Pitfall 5)
  - url: format!("{}/{}/{}", base, project, format!("_workitems/edit/{}", item.id))   — browser URL, NOT WIQL url field
  - labels: item.fields.tags.map(|t| t.split("; ").map(|s| s.to_string()).collect()).unwrap_or_default()
  - updated_at: item.fields.changed_date
- Return Ok(results)
