HELPER fn make_basic_auth(email: &str, api_token: &str) -> String:
- credentials = format!("{}:{}", email, api_token)
- format!("Basic {}", base64::engine::general_purpose::STANDARD.encode(credentials.as_bytes()))

HELPER fn extract_body(description: Option<serde_json::Value>) -> Option<String>:
- description.map(|adf| jc_adf::from_adf::to_markdown(&adf))
- Note: to_markdown() is infallible — no ? needed

FUNCTION validate_and_store(project_id: i32, site_url: &str, email: &str, api_token: &str, project_key: &str, project_path: &str, app_state: &crate::db::AppState) -> Result<String, String>:
- let base = normalize_instance_url(site_url);  — normalize once, reuse for all calls in this function
- let auth = make_basic_auth(email, api_token);
- Build reqwest::Client with 15s timeout
- GET format!("{}/rest/api/3/myself", base) with headers: Authorization: auth, Accept: application/json
- HTTP error handling: 401 → "Jira Cloud: bad credentials"; other → "Jira Cloud API error {status}: {reason}"
- Deserialize as JiraMyselfResponse
- display_name = user.display_name.or(user.email_address).unwrap_or_else(|| "unknown".to_string())
- Save TicketingConfig { provider: Some(ProviderConfig::Jiracloud(JiraCloudConfig { site_url: base.clone(), email: email.to_string(), project_key: project_key.to_string() })), updated_at: now_rfc3339() }
- Store StoredToken { access_token: api_token.to_string(), refresh_token: None, expires_at: None, provider: "jira_cloud".to_string() }
- Return Ok(display_name)

FUNCTION fetch_issues(site_url: &str, email: &str, api_token: &str, project_key: &str) -> Result<Vec<RemoteIssue>, String>:
- let base = normalize_instance_url(site_url);
- let auth = make_basic_auth(email, api_token);
- Build reqwest::Client with 15s timeout
- GET "{base}/rest/api/3/search" with query params: jql = "project = {project_key} AND statusCategory != Done ORDER BY updated DESC", maxResults = "100", fields = "summary,description,labels,updated,self"
  - Use .query(&[("jql", jql), ("maxResults", "100"), ("fields", "summary,description,labels,updated,self")])
- HTTP error handling same as validate_and_store
- Deserialize as JiraSearchResponse
- Map each issue to RemoteIssue:
  - external_id: format!("jira:{}", issue.key)
  - title: issue.fields.summary
  - body: extract_body(issue.fields.description)
  - url: format!("{}/browse/{}", base, issue.key)   — human-navigable URL, NOT issue.fields.self_url
  - labels: issue.fields.labels
  - updated_at: issue.fields.updated
- Return Ok(remote_issues)
