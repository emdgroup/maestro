   /// Public spawner: routes to local OR remote based on GitConnection
   pub async fn spawn_agent_execution(
       git_conn: &GitConnection,
       worktree: &Worktree,
       task: &Task,
       config: &ExecutionConfig,
   ) -> Result<ProcessOutput, String> {
       match git_conn {
           GitConnection::Local { path } => {
               spawn_agent_execution_local(path, worktree, task, config).await
           }
           GitConnection::Remote { ssh, remote_path } => {
               // Build remote worktree path: remote_path/worktree_name
               let remote_worktree = format!("{}/{}", remote_path, worktree.name);

               let handle = remote::spawn_remote_agent_execution(
                   ssh,
                   &remote_worktree,
                   worktree,
                   task,
                   config,
               ).await?;

               // Return ProcessOutput with remote_pid for tracking
               Ok(ProcessOutput {
                   success: true,
                   stdout: String::new(),
                   stderr: String::new(),
                   remote_pid: Some(handle.remote_pid),
                   is_remote: true,
               })
           }
       }
   }
