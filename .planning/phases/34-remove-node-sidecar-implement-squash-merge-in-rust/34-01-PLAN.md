
Step 3 — `git status --porcelain` to detect conflicts:
```rust
let status_output = TokioCommand::new("git")
    .args(["status", "--porcelain"])
    .current_dir(repo_path)
    .output()
    .await
    .map_err(|e| format!("Failed to run git status: {}", e))?;
let status_stdout = String::from_utf8_lossy(&status_output.stdout);

```

Step 4a — If conflicts detected, abort and return:

2. A `fn parse_conflict_files` helper (private, not pub) that parses `git status --porcelain` output for conflict markers:
```rust
fn parse_conflict_files(porcelain_status: &str) -> Vec<String> {
    porcelain_status
        .lines()
        .filter_map(|line| {
            if line.len() < 4 {
                return None;
            }
            let xy = &line[..2];
            // Conflict XY codes: any line where X or Y is 'U', plus AA and DD
            let is_conflict = xy.contains('U') || xy == "AA" || xy == "DD";
            if is_conflict {
                Some(line[3..].to_string())
            } else {
                None
            }
        })
        .collect()
}
```
