```typescript
import { test, expect } from '@playwright/test';

test.describe('Visual Regression Tests', () => {
  test('ProjectPicker renders in light mode', async ({ page }) => {
    await page.goto('/');
    await page.waitForLoadState('networkidle');

    // Verify ProjectPicker is visible
    const picker = page.locator('[data-testid="project-picker"]');
    await expect(picker).toBeVisible();

    // Capture baseline screenshot
    await expect(page).toHaveScreenshot('projectpicker-light.png');
  });

  test('ProjectPicker renders in dark mode', async ({ page }) => {
    await page.goto('/');
    // Force dark mode
    await page.evaluate(() => {
      document.documentElement.classList.add('dark');
      localStorage.setItem('theme_preference', 'dark');
    });
    await page.waitForLoadState('networkidle');

    const picker = page.locator('[data-testid="project-picker"]');
    await expect(picker).toBeVisible();
    await expect(page).toHaveScreenshot('projectpicker-dark.png');
  });

  test('AppHeader with tabs renders correctly', async ({ page }) => {
    // Navigate to main app (need project selected)
    // Option 1: Mock project selection in localStorage
    await page.evaluate(() => {
      localStorage.setItem('project_path', '/tmp/test-project');
    });

    await page.goto('/');
    await page.waitForLoadState('networkidle');

    const header = page.locator('header');
    await expect(header).toBeVisible();

    // Verify header height is compact (h-12 = 48px)
    const box = await header.boundingBox();
    expect(box?.height).toBeLessThanOrEqual(50);

    await expect(header).toHaveScreenshot('header-with-tabs.png');
  });

  test('Kanban board renders without layout shifts', async ({ page }) => {
    // Setup project
    await page.evaluate(() => {
      localStorage.setItem('project_path', '/tmp/test-project');
    });

    await page.goto('/');
    await page.waitForLoadState('networkidle');

    // Click Tasks tab to ensure kanban is visible
    const tasksTab = page.locator('button:has-text("Tasks")');
    await tasksTab.click();

    // Wait for kanban to render
    await page.waitForSelector('[data-testid="kanban-board"]', { timeout: 5000 });

    const kanban = page.locator('[data-testid="kanban-board"]');
    await expect(kanban).toHaveScreenshot('kanban-board.png');
  });

  test('No cumulative layout shift during startup', async ({ page }) => {
    await page.goto('/');

    const cls = await page.evaluate(() => {
      return new Promise<number>((resolve) => {
        let maxCLS = 0;
        new PerformanceObserver((list) => {
          for (const entry of list.getEntries()) {
            if ('hadRecentInput' in entry && !entry.hadRecentInput) {
              maxCLS += (entry as any).value;
            }
          }
        }).observe({ type: 'layout-shift', buffered: true });

        // Stop collecting after 3 seconds
        setTimeout(() => resolve(maxCLS), 3000);
      });
    });

    expect(cls).toBeLessThan(0.1); // WCAG passes < 0.1
  });

  test('Theme toggle does not cause layout shift', async ({ page }) => {
    await page.goto('/');
    await page.waitForLoadState('networkidle');

    // Get initial CLS
    const clsBefore = await page.evaluate(() => {
      return new Promise<number>((resolve) => {
        let sum = 0;
        new PerformanceObserver((list) => {
          for (const entry of list.getEntries()) {
            if ('hadRecentInput' in entry && !entry.hadRecentInput) {
              sum += (entry as any).value;
            }
          }
        }).observe({ type: 'layout-shift', buffered: true });
        setTimeout(() => resolve(sum), 1000);
      });
    });

    // Toggle dark mode
    await page.evaluate(() => {
      document.documentElement.classList.toggle('dark');
    });

    await page.waitForTimeout(500);

    // Check for additional layout shift
    const clsAfter = await page.evaluate(() => {
      return new Promise<number>((resolve) => {
        let sum = 0;
        new PerformanceObserver((list) => {
          for (const entry of list.getEntries()) {
            if ('hadRecentInput' in entry && !entry.hadRecentInput) {
              sum += (entry as any).value;
            }
          }
        }).observe({ type: 'layout-shift', buffered: true });
        setTimeout(() => resolve(sum), 1000);
      });
    });

    // Additional shift from theme toggle should be minimal
    expect(clsAfter - clsBefore).toBeLessThan(0.05);
  });
});
```
