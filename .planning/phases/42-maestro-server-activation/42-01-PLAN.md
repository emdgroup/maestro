use std::cell::RefCell;
use std::collections::HashMap;
use std::rc::Rc;
use agent_client_protocol::ClientSideConnection;
use tokio::sync::oneshot;

/// Handle for a single managed terminal subprocess
pub struct TerminalHandle {
    /// Accumulated output for ACP terminal/output polling
    pub output_buf: Rc<RefCell<String>>,
    /// Output byte limit from CreateTerminalRequest (None = unlimited)
    pub output_byte_limit: Option<u64>,
    /// Exit status set when subprocess exits
    pub exit_status: Rc<RefCell<Option<TerminalExitInfo>>>,
    /// Kill signal — dropping sender kills the background reader task
    pub kill_tx: Option<oneshot::Sender<()>>,
}

/// Exit information for a completed terminal
pub struct TerminalExitInfo {
    pub exit_code: Option<u32>,
    pub signal: Option<String>,
}

/// A live ACP session: maps maestro session_id to ACP internals
pub struct ActiveSession {
    /// The ACP SDK connection to the agent subprocess
    pub conn: ClientSideConnection,
    /// ACP-assigned session ID (from NewSessionResponse)
    pub acp_session_id: agent_client_protocol::SessionId,
    /// Managed terminals for this session
    pub terminals: Rc<RefCell<HashMap<String, TerminalHandle>>>,
    /// Agent child process handle (held to keep kill_on_drop active)
    pub child: tokio::process::Child,
}

/// Map from maestro session_id (String) to ActiveSession
pub type SessionMap = HashMap<String, ActiveSession>;

**2. `maestro-server/src/client.rs`** -- MaestroServerClient implementing acp::Client:

The struct holds:
- `stdout: Rc<tokio::sync::Mutex<tokio::io::Stdout>>` -- shared stdout writer (Rc not Arc because !Send context)
- `pending_permissions: Rc<RefCell<HashMap<String, oneshot::Sender<agent_client_protocol::RequestPermissionResponse>>>>` -- permission oneshot map
- `terminals: Rc<RefCell<HashMap<String, TerminalHandle>>>` -- shared with ActiveSession
- `maestro_session_id: String` -- for tagging response frames
- `terminal_counter: Rc<RefCell<u64>>` -- monotonic terminal ID counter

Implement `#[async_trait::async_trait(?Send)] impl agent_client_protocol::Client for MaestroServerClient`:

**`session_notification`** (SERVER-02):
- Serialize the `SessionNotification` args with `serde_json::to_value(&args)` into the `payload` field
- Build `MaestroRpcMessage::Response(ServerResponse::SessionUpdate(SessionUpdate { session_id: self.maestro_session_id.clone(), payload }))`
- Call `send_response(&self.stdout, &msg).await`
- Return `Ok(())`

**`request_permission`** (SERVER-04):
- Extract `request_id` from `args.tool_call.tool_call_id.to_string()`
- Create `oneshot::channel()`
- Insert sender into `self.pending_permissions.borrow_mut()` keyed by `request_id.clone()`
- Serialize args to `serde_json::to_value(&args)` as payload
- Build `MaestroRpcMessage::Response(ServerResponse::PermissionRequest(maestro_protocol::PermissionRequest { session_id: self.maestro_session_id.clone(), request_id, payload }))`
- Call `send_response(&self.stdout, &msg).await`
- CRITICAL: `drop` the stdout guard is not needed here since `send_response` is a separate function that acquires and releases its own lock
- `rx.await.map_err(|_| agent_client_protocol::Error::internal_error().with_message("permission channel closed".to_string()))`

**`create_terminal`** (SERVER-03):
- Increment terminal_counter, generate `terminal_id = format!("term-{}", counter)`
- Spawn subprocess: `tokio::process::Command::new(&args.command).args(&args.args).current_dir(args.cwd.as_deref().unwrap_or(std::path::Path::new("."))).stdout(Stdio::piped()).stderr(Stdio::piped()).kill_on_drop(true).spawn()`
- If spawn fails, return `Err(acp::Error::internal_error().with_message(e.to_string()))`
- Take stdout pipe from child
- Create `output_buf: Rc<RefCell<String>>`, `exit_status: Rc<RefCell<Option<TerminalExitInfo>>>`
- Create `(kill_tx, kill_rx) = oneshot::channel()`
- Clone stdout Rc, buf Rc, session_id, terminal_id for the spawned task
- `tokio::task::spawn_local` a background reader that:
  - Uses `tokio::io::BufReader` on the stdout pipe
  - Reads lines in a `tokio::select!` loop (biased toward kill_rx)
  - Appends each line to `output_buf` (respecting `output_byte_limit` -- truncate from front when exceeded)
  - Writes `ServerResponse::TerminalOutput` frame to stdout for each chunk
  - On EOF: sets exit_status from child.wait() if available, breaks
  - On kill_rx: breaks
- Build `TerminalHandle { output_buf, output_byte_limit: args.output_byte_limit, exit_status, kill_tx: Some(kill_tx) }`
- Insert into `self.terminals.borrow_mut()`
- Return `Ok(CreateTerminalResponse::new(terminal_id))`

**`terminal_output`**:
- Look up terminal by `args.terminal_id.to_string()` in `self.terminals.borrow()`
- Return accumulated `output_buf.borrow().clone()`, `truncated: false` (or check byte limit), and `exit_status` if set
- If terminal not found, return `Err(Error::internal_error().with_message("unknown terminal"))`

**`release_terminal`**:
- Remove terminal from map by `args.terminal_id.to_string()`
- Drop the `TerminalHandle` (kill_tx drop kills background reader; child subprocess killed by kill_on_drop)
- Return `Ok(ReleaseTerminalResponse::new())`

**`kill_terminal`**:
- Look up terminal, take `kill_tx` out (`Option::take`), send on it to kill the reader
- Return `Ok(KillTerminalResponse::new())`

**`wait_for_terminal_exit`**:
- Look up terminal, poll `exit_status` -- if already set return it
- If not set, create a polling loop with `tokio::time::sleep(Duration::from_millis(100))` until exit_status is set, then return it
- Return `WaitForTerminalExitResponse::new(TerminalExitStatus::new().exit_code(info.exit_code).signal(info.signal))`

**Default implementations for `write_text_file`, `read_text_file`, `ext_method`, `ext_notification`**: use the default trait implementations (return `Err(Error::method_not_found())` for fs ops, `Ok(...)` for ext).

Add a public helper function:

```rust
pub async fn send_response(
    stdout: &Rc<tokio::sync::Mutex<tokio::io::Stdout>>,
    msg: &maestro_protocol::MaestroRpcMessage,
) -> std::result::Result<(), Box<dyn std::error::Error>> {
    use tokio::io::AsyncWriteExt;
    let mut buf: Vec<u8> = Vec::new();
    maestro_protocol::write_message(&mut buf, msg).await?;
    let mut out = stdout.lock().await;
    out.write_all(&buf).await?;
    out.flush().await?;
    Ok(())
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| stdin (Tauri host -> maestro-server) | All incoming messages are untrusted; must be deserialized via typed serde |
| stdout (maestro-server -> Tauri host) | Output frames are constructed internally; no injection risk |
| agent subprocess -> client callbacks | Agent can trigger arbitrary terminal commands via create_terminal |
