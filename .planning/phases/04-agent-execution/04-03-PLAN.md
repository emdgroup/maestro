```rust
#[tauri::command]
pub fn get_execution_logs(
    app_state: State<'_, Arc<AppState>>,
    task_id: i32,
) -> Result<Vec<gsd_demo::models::ExecutionLog>, String> {
    let conn = app_state.db.lock().map_err(|e| format!("Lock failed: {}", e))?;

    let mut stmt = conn.prepare(
        "SELECT id, task_id, status, output, started_at, completed_at
         FROM execution_logs
         WHERE task_id = ?
         ORDER BY started_at DESC"
    ).map_err(|e| e.to_string())?;

    let logs = stmt.query_map(rusqlite::params![task_id], |row| {
        Ok(gsd_demo::models::ExecutionLog {
            id: row.get(0)?,
            task_id: row.get(1)?,
            status: row.get(2)?,
            output: row.get(3)?,
            started_at: row.get(4)?,
            completed_at: row.get(5)?,
        })
    }).map_err(|e| e.to_string())?;

    let mut result = Vec::new();
    for log in logs {
        result.push(log.map_err(|e| e.to_string())?);
    }
    Ok(result)
}
```

Add executeTask action to store for triggering agent execution.

In src/store/boardStore.ts, add to the actions section:

```typescript
executeTask: async (projectId: number, taskId: number, repoPath: string) => {
  try {
    // Invoke spawn_agent_execution handler
    const executionLogId = await invoke<number>("spawn_agent_execution", {
      project_id: projectId,
      task_id: taskId,
      repo_path: repoPath,
    });

    // Update task status to InProgress
    set((state) =>
      produce(state, (draft) => {
        const task = draft.tasks.find((t) => t.id === taskId);
        if (task) {
          task.status = "InProgress";
          task.execution_log_id = executionLogId;  // Store execution_log id
        }
      })
    );

    return executionLogId;
  } catch (error) {
    console.error("Execute task failed:", error);
    throw error;
  }
}
```

Key:
- Calls spawn_agent_execution IPC handler
- Returns execution_log_id from handler
- Updates task status to InProgress
- Stores execution_log_id for later history retrieval
- Error thrown for caller to handle
