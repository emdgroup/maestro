  <name>Task 3: Implement spawn_agent_execution IPC handler</name>
  <files>src-tauri/src/ipc/handlers.rs</files>
  <action>
Implement spawn_agent_execution handler that creates execution log and spawns background task.

Add to src-tauri/src/ipc/handlers.rs:

```rust
#[tauri::command]
pub async fn spawn_agent_execution(
    app_state: State<'_, Arc<AppState>>,
    project_id: i32,
    task_id: i32,
    repo_path: String,
) -> Result<i32, String> {
    // 1. Create execution log record
    let exec_log_id = {
        let conn = app_state.db.lock().map_err(|e| format!("Lock failed: {}", e))?;
        crate::db::execution_logs::create_execution_log(&conn, task_id, 0)?
    };

    // 2. Lease worktree from pool (will implement in 04-03)
    // For now, use a placeholder worktree path
    let worktree_path = format!("{}/pool/wt-001", repo_path);

    // 3. Clone app_state for background task
    let app_state_clone = app_state.clone();
    let repo_clone = repo_path.clone();

    // 4. Spawn background task (returns immediately)
    tokio::spawn(async move {
        // Run agent process via sidecar
        match crate::process::spawner::spawn_agent_cli(
            &worktree_path,
            "sidecar/dist/index.js",  // Path to compiled sidecar
            task_id,
        ).await {
            Ok(output) => {
                // Append output to execution log
                {
                    let conn = app_state_clone.db.lock().unwrap();
                    let _ = crate::db::execution_logs::append_output(&conn, exec_log_id, &output.stdout);
                    if !output.stderr.is_empty() {
                        let _ = crate::db::execution_logs::append_output(&conn, exec_log_id, &format!("\n[STDERR]\n{}", output.stderr));
                    }
                    let _ = crate::db::execution_logs::mark_complete(&conn, exec_log_id, output.exit_code);
                }
            }
            Err(e) => {
                // Log error to execution log
                let conn = app_state_clone.db.lock().unwrap();
                let _ = crate::db::execution_logs::append_output(&conn, exec_log_id, &format!("\n[ERROR] {}", e));
                let _ = crate::db::execution_logs::mark_complete(&conn, exec_log_id, -1);
            }
        }
    });

    // 5. Return execution_log id immediately (process runs in background)
    Ok(exec_log_id)
}
```

Pattern reference: 04-RESEARCH.md Pattern 3 (lines 565-608).

Key points:
- Function is async (allows tokio::spawn inside)
- Lock database only when needed (bracket scope)
- Clone app_state for background task closure
- Return immediately with exec_log_id
- Background task continues after IPC returns
- Error handling captures errors to execution_log
