    ```typescript
    import { simpleGit, MergeResult } from "simple-git";
    import path from "path";

    export interface MergeOutcome {
      success: boolean;
      conflicts: string[];
      conflictFiles?: string[];
      mergeCommitSha?: string;
      message?: string;
    }

    export async function squashMergeToMain(
      repoPath: string,
      taskId: number,
      taskBranchName: string,
      taskName: string
    ): Promise<MergeOutcome> {
      const git = simpleGit(repoPath);

      try {
        // Ensure we're on main branch
        await git.checkout("main");

        // Attempt squash merge (--no-commit to control commit message)
        const result: MergeResult = await git.merge([
          taskBranchName,
          "--squash",
          "--no-commit",
        ]);

        if (result.failed) {
          // Merge conflict detected
          return {
            success: false,
            conflicts: result.conflicts || [],
            conflictFiles: result.conflicted || [],
            message: "Merge conflict detected. Aborting merge.",
          };
        }

        // Create merge commit with task reference
        const commitMsg = `Merge task #${taskId}: ${taskName}

All agent commits squashed into single commit.`;

        const commitResult = await git.commit(commitMsg);

        // Get commit SHA for verification
        const logResult = await git.log(["-1", "--format=%H"]);
        const commitSha = logResult.latest?.hash || "unknown";

        return {
          success: true,
          conflicts: [],
          mergeCommitSha: commitSha,
          message: "Merge successful",
        };
      } catch (error) {
        // Non-conflict merge error (permission, branch not found, etc.)
        return {
          success: false,
          conflicts: [`Merge error: ${error.message}`],
          message: error.message,
        };
      }
    }
    ```

    Add function: `abortMergeOnConflict`
    ```typescript
    export async function abortMergeOnConflict(
      repoPath: string
    ): Promise<boolean> {
      const git = simpleGit(repoPath);

      try {
        await git.merge(["--abort"]);
        return true;
      } catch (error) {
        console.error("Failed to abort merge:", error);
        return false;
      }
    }
    ```

    Export both functions in sidecar/src/index.ts for use by Rust backend.

    #[tauri::command]
    pub async fn approve_task_and_merge(
        app_state: State<Arc<AppState>>,
        app: tauri::AppHandle,
        task_id: i32,
    ) -> Result<serde_json::json, String> {
        let conn = app_state.db.lock().map_err(|e| format!("Lock failed: {}", e))?;
        let now = chrono::Utc::now().to_rfc3339();

        // Query task details
        let (task_name, branch_name, worktree_path, project_id): (String, String, String, i32) = conn
            .query_row(
                "SELECT t.name, w.branch_name, w.path, t.project_id
                 FROM tasks t
                 JOIN worktrees w ON w.task_id = t.id
                 WHERE t.id = ?",
                [task_id],
                |row| Ok((row.get(0)?, row.get(1)?, row.get(2)?, row.get(3)?)),
            )
            .map_err(|e| format!("Task not found: {}", e))?;

        // Get project repo path
        let repo_path: String = conn
            .query_row(
                "SELECT path FROM projects WHERE id = ?",
                [project_id],
                |row| row.get(0),
            )
            .map_err(|e| format!("Project not found: {}", e))?;

        // Update task status to "Merging" (transient state)
        conn.execute(
            "UPDATE tasks SET status = 'Merging', updated_at = ? WHERE id = ?",
            rusqlite::params![now, task_id],
        )
        .map_err(|e| format!("Failed to update task status: {}", e))?;

        drop(conn);  // Release lock before async operation

        // Spawn async merge operation in background
        let app_clone = app.clone();
        let app_state_clone = app_state.inner().clone();
        tokio::spawn(async move {
            // Call Node.js sidecar to perform squash merge
            // (Exact invocation depends on sidecar integration pattern from Phase 3/5)
            // Pseudo-code:
            match invoke_sidecar_merge(
                &repo_path,
                task_id,
                &branch_name,
                &task_name,
            ).await {
                Ok(merge_result) if merge_result.success => {
                    // Merge successful: update task to Done, cleanup worktree
                    if let Err(e) = finalize_successful_merge(
                        &app_state_clone,
                        task_id,
                        &worktree_path,
                    ).await {
                        eprintln!("Merge finalization error: {}", e);
                        let _ = app_clone.emit("merge_error", json!({
                            "task_id": task_id,
                            "error": format!("Merge succeeded but cleanup failed: {}", e)
                        }));
                    } else {
                        let _ = app_clone.emit("merge_complete", json!({
                            "task_id": task_id,
                            "status": "Done"
                        }));
                    }
                }
                Ok(merge_result) => {
                    // Merge failed or conflicts detected
                    if !merge_result.conflicts.is_empty() {
                        // Conflict: reject task back to Review with conflict feedback
                        let _ = reject_merge_on_conflict(
                            &app_state_clone,
                            task_id,
                            &merge_result.conflicts,
                        ).await;
                    }
                    let _ = app_clone.emit("merge_error", json!({
                        "task_id": task_id,
                        "conflicts": merge_result.conflicts,
                        "message": merge_result.message
                    }));
                }
                Err(e) => {
                    // System error
                    let _ = app_clone.emit("merge_error", json!({
                        "task_id": task_id,
                        "error": e.to_string()
                    }));
                }
            }
        });

        Ok(json!({ "merging": true, "message": "Merge started" }))
    }

    // Helper function: finalize successful merge
    async fn finalize_successful_merge(
        app_state: &Arc<AppState>,
        task_id: i32,
        worktree_path: &str,
    ) -> Result<(), String> {
        let conn = app_state.db.lock().map_err(|e| format!("Lock failed: {}", e))?;
        let now = chrono::Utc::now().to_rfc3339();

        // Update task status to Done
        conn.execute(
            "UPDATE tasks SET status = 'Done', updated_at = ? WHERE id = ?",
            rusqlite::params![now, task_id],
        )
        .map_err(|e| format!("Update task failed: {}", e))?;

        // Return worktree to pool (update worktree status to Available)
        conn.execute(
            "UPDATE worktrees SET task_id = NULL, status = 'Available', updated_at = ?
             WHERE path = ?",
            rusqlite::params![now, worktree_path],
        )
        .map_err(|e| format!("Return worktree failed: {}", e))?;

        Ok(())
    }

    // Helper function: reject merge on conflict
    async fn reject_merge_on_conflict(
        app_state: &Arc<AppState>,
        task_id: i32,
        conflicts: &[String],
    ) -> Result<(), String> {
        let conn = app_state.db.lock().map_err(|e| format!("Lock failed: {}", e))?;
        let now = chrono::Utc::now().to_rfc3339();
        let conflict_feedback = format!("Merge conflict:\n{}", conflicts.join("\n"));

        // Update task status back to Review (not InProgress, stays in Review with conflict message)
        conn.execute(
            "UPDATE tasks SET status = 'Review', updated_at = ? WHERE id = ?",
            rusqlite::params![now, task_id],
        )
        .map_err(|e| format!("Update task failed: {}", e))?;

        // Save conflict feedback as review comment (for visibility)
        conn.execute(
            "INSERT INTO task_reviews (task_id, decision, general_feedback, created_at)
             VALUES (?, 'RequestChanges', ?, ?)",
            rusqlite::params![task_id, conflict_feedback, now],
        )
        .map_err(|e| format!("Save feedback failed: {}", e))?;

        Ok(())
    }
    ```

    import { toast } from "sonner";

    useEffect(() => {
      let unlistenComplete: (() => void) | null = null;
      let unlistenError: (() => void) | null = null;

      (async () => {
        // Listen for merge_complete event
        unlistenComplete = await listen("merge_complete", (event: Event<any>) => {
          const { task_id, status } = event.payload;
          toast.success(`Merge successful! Task moved to Done.`);
          // Refresh board
          loadTasks();
        });

        // Listen for merge_error event
        unlistenError = await listen("merge_error", (event: Event<any>) => {
          const { task_id, conflicts, error, message } = event.payload;
          if (conflicts && conflicts.length > 0) {
            toast.error(`Merge conflict: ${conflicts.join(", ")}`);
          } else {
            toast.error(`Merge failed: ${error || message}`);
          }
          // Refresh board
          loadTasks();
        });
      })();

      return () => {
        if (unlistenComplete) unlistenComplete();
        if (unlistenError) unlistenError();
      };
    }, []);
    ```
