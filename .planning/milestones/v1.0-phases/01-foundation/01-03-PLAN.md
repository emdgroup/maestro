   use ts_rs::TS;

   #[derive(Debug, Clone, Serialize, Deserialize, TS)]
   #[ts(export)]
   pub struct Project {
       pub id: i32,
       pub name: String,
       pub path: String,
       pub created_at: String,  // ISO 8601
   }

   #[derive(Debug, Clone, Serialize, Deserialize, TS)]

       Archived,
   }
