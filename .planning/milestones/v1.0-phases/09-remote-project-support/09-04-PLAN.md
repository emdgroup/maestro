   ```rust
   #[tauri::command]
   pub async fn test_remote_connection(
       config: SshConfig,
       state: State<'_, AppState>,
   ) -> Result<bool, String> {
       // 1. Create temporary RemoteSshSession from config
       let session = RemoteSshSession::new(config);

       // 2. Attempt connect with 10s timeout
       // 3. If successful, return Ok(true)
       // 4. If auth fails, return Err("Authentication failed: ...")
       // 5. If network fails, return Err("Connection timeout: ...")
       // 6. Disconnect after test

       session.connect().await.map(|_| {
           let _ = tokio::spawn(async move {
               let _ = session.disconnect().await;
           });
           true
       }).map_err(|e| format!("Connection test failed: {:?}", e))
   }

   #[tauri::command]
   pub async fn get_remote_connection_status(
       project_id: i64,
       state: State<'_, AppState>,
   ) -> Result<ConnectionStatus, String> {
       let session = state.get_ssh_session(project_id).await;
       let is_connected = if let Some(s) = session {
           s.is_connected().await
       } else {
           false
       };

       Ok(ConnectionStatus {
           project_id,
           is_remote: true,
           connected: is_connected,
           disconnected_reason: if !is_connected { Some("Not connected".into()) } else { None },
       })
   }

   #[tauri::command]
   pub async fn reconnect_remote_project(
       project_id: i64,
       state: State<'_, AppState>,
   ) -> Result<(), String> {
       let project = get_project(project_id, &state)?;
       if !project.is_remote {
           return Err("Project is not remote".into());
       }

       let config = project.ssh_config.as_ref().ok_or("No SSH config")?;
       let session = RemoteSshSession::new(config.clone());
       session.connect().await.map_err(|e| format!("Reconnection failed: {:?}", e))?;

       state.set_ssh_session(project_id, session).await;
       Ok(())
   }
   ```

   ```typescript
   export function ProjectPicker({ onProjectSelected }: Props) {

     const [localPath, setLocalPath] = useState('');
     const [remoteConfig, setRemoteConfig] = useState<SshConfig | null>(null);

     const handleSelectLocal = () => setStage('local');
     const handleSelectRemote = () => setStage('remote');

     const handleCreateLocal = async () => {

         name: localPath.split('/').pop() || 'project',
         path: localPath,
         isRemote: false,
       });
       onProjectSelected(project);
     };

     const handleCreateRemote = async (config: SshConfig) => {
       // Test connection first (user has already tested, but verify again)

       // Create project with remote config

         name: config.host,
         path: '',

       });

       onProjectSelected(project);
     };

     if (stage === 'select') {
       return (
         <div className="project-picker">
           <h2>Create Project</h2>
           <button onClick={handleSelectLocal}>📁 Local Project</button>

         </div>
       );

     if (stage === 'local') {
       return (
         <div className="project-picker">
           <h2>Local Project Path</h2>
   ```

1. Update ProjectCard.tsx to show remote badge and connection status:
   ```typescript
   import { useEffect, useState } from 'react';
   import { invoke } from '@tauri-apps/api/core';

   export function ProjectCard({ project }: Props) {
     const [connectionStatus, setConnectionStatus] = useState<ConnectionStatus | null>(null);

     useEffect(() => {
       if (!project.isRemote) return;

       // Check connection status on component mount
       const checkStatus = async () => {
         try {
           const status = await invoke('get_remote_connection_status', {
             projectId: project.id,
           });
           setConnectionStatus(status);
         } catch (error) {
           console.error('Failed to get connection status:', error);
         }
       };

       checkStatus();
       const interval = setInterval(checkStatus, 10000);  // Poll every 10s
       return () => clearInterval(interval);
     }, [project.isRemote, project.id]);

     const handleRetryConnection = async () => {
       if (!project.isRemote) return;
       try {
         await invoke('reconnect_remote_project', { projectId: project.id });
         const status = await invoke('get_remote_connection_status', {
           projectId: project.id,
         });
         setConnectionStatus(status);
       } catch (error) {
         toast.error(`Reconnection failed: ${error}`);
       }
     };

     return (
       <div className="project-card">
         <h3>{project.name}</h3>

         {/* Remote badge and connection indicator */}
         {project.isRemote && (
           <div className="remote-indicator">
             <span className="badge">🌐 Remote</span>
             <span className={`status ${connectionStatus?.connected ? 'connected' : 'disconnected'}`}>
               {connectionStatus?.connected ? '✓ Connected' : '✗ Disconnected'}
             </span>
             {!connectionStatus?.connected && (
               <button
                 className="retry-btn"
                 onClick={handleRetryConnection}
                 title="Retry connection"
               >
                 🔄
               </button>
             )}
           </div>
         )}

         {/* Rest of project card UI */}
         <p className="path">{project.path || project.sshConfig?.remotePath}</p>
       </div>
     );
   }
   ```

   ```typescript
     // ... existing state ...
     remoteConnectionStatus: Map<number, ConnectionStatus>;  // project_id -> status

   interface BoardActions {

     getRemoteConnectionStatus: (projectId: number) => ConnectionStatus | undefined;

   export const useBoardStore = create<BoardState & BoardActions>((set, get) => ({
     // ... existing state ...
     remoteConnectionStatus: new Map(),

     updateRemoteConnectionStatus: (projectId, status) =>
       set((state) => ({
         remoteConnectionStatus: new Map(state.remoteConnectionStatus)
           .set(projectId, status),
       })),

     getRemoteConnectionStatus: (projectId) =>
   ```
