Example:
```typescript
resumeExecution: async (projectId: number, taskId: number, repoPath: string) => {
  set(state => ({ retryingTaskIds: new Set(state.retryingTaskIds).add(taskId) }));
  try {
    const logId = await invoke("spawn_agent_execution", { projectId, taskId, repoPath });
    // Update task status to InProgress (if not already)
    return logId;
  } catch (error) {
    throw error;
  } finally {
    set(state => {
      const ids = new Set(state.retryingTaskIds);
      ids.delete(taskId);
      return { retryingTaskIds: ids };
    });
  }
}
```
